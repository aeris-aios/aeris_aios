app.get("/api/files/:projectId", requireAuth, async (req, res) => {
  const projectDir = path.join(PROJECTS_DIR, req.session.userId, req.params.projectId);
  async function buildTree(dir, prefix = "") {
    let entries;
    try { entries = await fs.readdir(dir, { withFileTypes: true }); } catch { return []; }
    const result = [];
    const skip = new Set([".git", "node_modules", "__pycache__", ".next", "dist", ".venv"]);
    for (const e of entries.sort((a, b) => {
      if (a.isDirectory() && !b.isDirectory()) return -1;
      if (!a.isDirectory() && b.isDirectory()) return 1;
      return a.name.localeCompare(b.name);
    })) {
      if (skip.has(e.name) || e.name.startsWith(".")) continue;
      const rel = path.join(prefix, e.name);
      if (e.isDirectory()) {
        result.push({ name: e.name, path: rel, type: "dir", children: await buildTree(path.join(dir, e.name), rel) });
      } else {
        result.push({ name: e.name, path: rel, type: "file" });
      }
    }
    return result;
  }
  res.json(await buildTree(projectDir));
});

    res.end();
  }

    send({ stage: "done", message: "Content generated!", content: result });

    send({ stage: "error", message: err.message });
  }

const Anthropic = require("@anthropic-ai/sdk");
const fs = require("fs/promises");
const path = require("path");
const { execSync } = require("child_process");

const TOOLS = [
  { name: "read_file", description: "Read a file from the project.", input_schema: { type: "object", properties: { file_path: { type: "string", description: "Relative path to the file" } }, required: ["file_path"] } },
  { name: "write_file", description: "Create or overwrite a file.", input_schema: { type: "object", properties: { file_path: { type: "string", description: "Relative path" }, content: { type: "string", description: "File content" } }, required: ["file_path", "content"] } },
  { name: "edit_file", description: "Replace specific text in a file.", input_schema: { type: "object", properties: { file_path: { type: "string" }, old_text: { type: "string" }, new_text: { type: "string" } }, required: ["file_path", "old_text", "new_text"] } },
  { name: "list_directory", description: "List files in a directory.", input_schema: { type: "object", properties: { directory: { type: "string" } }, required: ["directory"] } },
  { name: "search_files", description: "Search for text across files.", input_schema: { type: "object", properties: { pattern: { type: "string" }, file_glob: { type: "string" } }, required: ["pattern"] } },
  { name: "run_command", description: "Run a shell command.", input_schema: { type: "object", properties: { command: { type: "string" } }, required: ["command"] } },
  { name: "create_directory", description: "Create a directory.", input_schema: { type: "object", properties: { directory: { type: "string" } }, required: ["directory"] } },
  { name: "delete_file", description: "Delete a file.", input_schema: { type: "object", properties: { file_path: { type: "string" } }, required: ["file_path"] } },
];

const BLOCKED = [/rm\s+-rf\s+[/~]/, /mkfs\./, /dd\s+if=/, />\s*\/dev\/sd/, /curl.*\|\s*(bash|sh)/, /wget.*\|\s*(bash|sh)/];

function safePath(projectDir, rel) {
  const normalized = path.normalize(rel).replace(/^(\.\.[/\\])+/, "");
  const resolved = path.resolve(projectDir, normalized);
  if (!resolved.startsWith(path.resolve(projectDir))) throw new Error("Path outside project");
  return resolved;
}

async function executeTool(name, input, dir) {
  switch (name) {
    case "read_file": return await fs.readFile(safePath(dir, input.file_path), "utf-8");
    case "write_file": { const p = safePath(dir, input.file_path); await fs.mkdir(path.dirname(p), { recursive: true }); await fs.writeFile(p, input.content); return `Written: ${input.file_path}`; }
    case "edit_file": { const p = safePath(dir, input.file_path); const c = await fs.readFile(p, "utf-8"); if (!c.includes(input.old_text)) return "Error: text not found"; await fs.writeFile(p, c.replace(input.old_text, input.new_text)); return `Edited: ${input.file_path}`; }
    case "list_directory": { const entries = await fs.readdir(safePath(dir, input.directory), { withFileTypes: true }); return entries.map(e => `${e.isDirectory() ? "[dir] " : "[file] "}${e.name}`).join("\n"); }
    case "search_files": { try { return execSync(`grep -rn '${input.pattern.replace(/'/g, "'\\''")}' . --include='*' -l 2>/dev/null | head -50`, { cwd: dir, encoding: "utf-8", timeout: 15000 }) || "No matches."; } catch { return "No matches."; } }
    case "run_command": { for (const b of BLOCKED) if (b.test(input.command)) return "Blocked for security."; try { return execSync(input.command, { cwd: dir, encoding: "utf-8", timeout: 60000, maxBuffer: 1024*1024 }).slice(0, 15000) || "(no output)"; } catch (e) { return `Failed:\n${(e.stdout||"").slice(0,5000)}\n${(e.stderr||"").slice(0,5000)}`; } }
    case "create_directory": { await fs.mkdir(safePath(dir, input.directory), { recursive: true }); return `Created: ${input.directory}`; }
    case "delete_file": { await fs.unlink(safePath(dir, input.file_path)); return `Deleted: ${input.file_path}`; }
    default: return "Unknown tool";
  }
}

const SYS = `You are ATREYU, an AI coding assistant in the ATREYU AIOS IDE. You have full project access through tools. Be direct, write actual code, verify your work.`;

async function runAgent(apiKey, msg, dir, history, onEvent) {
  const client = new Anthropic({ apiKey });
  history.push({ role: "user", content: msg });
  let iter = 0;
  while (iter++ < 25) {
    const res = await client.messages.create({ model: "claude-sonnet-4-6", max_tokens: 8192, system: SYS, tools: TOOLS, messages: history });
    history.push({ role: "assistant", content: res.content });
    for (const b of res.content) if (b.type === "text") onEvent({ type: "text", text: b.text });
    if (res.stop_reason === "end_turn") break;
    const results = [];
    for (const b of res.content) {
      if (b.type === "tool_use") {
        onEvent({ type: "tool_use", tool: b.name, input: b.input });
        try {
          const r = await executeTool(b.name, b.input, dir);
          results.push({ type: "tool_result", tool_use_id: b.id, content: r });
          onEvent({ type: "tool_result", tool: b.name, result: r.slice(0, 500) });
        } catch (e) {
          results.push({ type: "tool_result", tool_use_id: b.id, content: `Error: ${e.message}`, is_error: true });
          onEvent({ type: "tool_error", tool: b.name, error: e.message });
        }
      }
    }
    if (results.length > 0) history.push({ role: "user", content: results });
    else break;
  }
  return history;
}

const Anthropic = require("@anthropic-ai/sdk");
const { execSync } = require("child_process");
const fs = require("fs/promises");
const path = require("path");
const { v4: uuidv4 } = require("uuid");

const CLONE_DIR = path.join(process.cwd(), "data", "repos");
const CODE_EXT = new Set([".js",".ts",".jsx",".tsx",".py",".rb",".go",".rs",".java",".c",".cpp",".h",".cs",".php",".swift",".kt",".scala",".vue",".svelte",".html",".css",".scss",".sql",".sh",".yaml",".yml",".toml",".json",".md",".mdx",".txt"]);
const SKIP = new Set(["node_modules",".git","__pycache__",".next","dist","build","vendor",".venv","venv","target","bin","obj",".cache","coverage"]);

async function cloneRepo(url) {
  await fs.mkdir(CLONE_DIR, { recursive: true });
  const name = path.basename(url, ".git").replace(/[^a-zA-Z0-9_-]/g, "-");
  const dest = path.join(CLONE_DIR, `${name}-${Date.now()}`);
  execSync(`git clone --depth 20 "${url}" "${dest}"`, { timeout: 120000, stdio: "pipe" });
  return dest;
}

async function collectFiles(dir, max = 60, maxSize = 8000) {
  const files = [];
  async function walk(d, prefix = "") {
    if (files.length >= max) return;
    let entries; try { entries = await fs.readdir(d, { withFileTypes: true }); } catch { return; }
    entries.sort((a, b) => { const p = n => { const l = n.toLowerCase(); if (l.startsWith("readme")) return 0; if (["package.json","pyproject.toml","cargo.toml"].includes(l)) return 1; if (l.includes("main")||l.includes("index")||l.includes("app")) return 3; return 10; }; return p(a.name)-p(b.name); });
    for (const e of entries) {
      if (files.length >= max) break;
      const full = path.join(d, e.name), rel = path.join(prefix, e.name);
      if (e.isDirectory()) { if (!SKIP.has(e.name) && !e.name.startsWith(".")) await walk(full, rel); }
      else { const ext = path.extname(e.name).toLowerCase(); if (CODE_EXT.has(ext) || ["makefile","dockerfile"].includes(e.name.toLowerCase())) { try { const st = await fs.stat(full); if (st.size > 100000) continue; let c = await fs.readFile(full, "utf-8"); if (c.length > maxSize) c = c.slice(0, maxSize) + "\n...(truncated)"; files.push({ path: rel, content: c }); } catch {} } }
    }
  }
  await walk(dir);
  return files;
}

async function buildTree(dir, depth = 0, max = 4) {
  if (depth > max) return "";
  let r = "", entries; try { entries = await fs.readdir(dir, { withFileTypes: true }); } catch { return r; }
  for (const e of entries) {
    if (SKIP.has(e.name) || e.name.startsWith(".")) continue;
    r += "  ".repeat(depth) + (e.isDirectory() ? e.name + "/\n" : e.name + "\n");
    if (e.isDirectory()) r += await buildTree(path.join(dir, e.name), depth + 1, max);
  }
  return r;
}

async function analyzeRepo(apiKey, repoDir, repoUrl, onProgress) {
  const client = new Anthropic({ apiKey });
  onProgress({ stage: "reading", message: "Reading repository files..." });
  const [files, tree] = await Promise.all([collectFiles(repoDir), buildTree(repoDir)]);
  onProgress({ stage: "analyzing", message: `Analyzing ${files.length} files with Claude...` });
  const ctx = files.map(f => `--- FILE: ${f.path} ---\n${f.content}\n`).join("\n");

  const res = await client.messages.create({
    model: "claude-sonnet-4-6", max_tokens: 8000,
    messages: [{ role: "user", content: `You are ATREYU, an AI that learns skills from codebases. Analyze this repo and extract distinct, reusable SKILLS.\n\nA "skill" is a capability, pattern, technique, or domain knowledge from this codebase.\n\n## Repository: ${repoUrl}\n\n## Structure:\n${tree.slice(0,3000)}\n\n## Files:\n${ctx}\n\nExtract 5-15 skills. Each skill needs:\n1. name (3-8 words)\n2. category: Frontend/Backend/Database/DevOps/API/Auth/Testing/AI-ML/Architecture/Tooling/Data-Processing/Security/Content-Strategy/Social-Media/Copywriting\n3. summary (under 120 chars)\n4. description (3-5 sentences)\n5. key_concepts (3-6 items)\n6. code_example (short snippet)\n7. use_cases (2-4 items)\n8. complexity: beginner/intermediate/advanced\n\nRespond with ONLY a JSON array. No markdown fences.` }],
  });

  onProgress({ stage: "processing", message: "Processing extracted skills..." });
  let skills;
  try { const t = res.content[0].text.trim().replace(/^```json?\s*/, "").replace(/\s*```$/, ""); skills = JSON.parse(t); } catch (e) { throw new Error("Failed to parse skills: " + e.message); }
  if (!Array.isArray(skills)) throw new Error("Expected array of skills");

  const now = new Date().toISOString();
  return skills.map(s => ({ id: uuidv4(), name: s.name || "Unnamed", category: s.category || "Architecture", summary: s.summary || "", description: s.description || "", key_concepts: s.key_concepts || [], code_example: s.code_example || "", use_cases: s.use_cases || [], complexity: s.complexity || "intermediate", source_repo: repoUrl, trained_at: now }));
}

async function cleanupRepo(dir) { try { await fs.rm(dir, { recursive: true, force: true }); } catch {} }

async function ensure() {
  try { await fs.access(FILE); } catch {
    await fs.mkdir(path.dirname(FILE), { recursive: true });
    await fs.writeFile(FILE, "[]");
  }
}

async function getAllContent() {
  await ensure();
  try { return JSON.parse(await fs.readFile(FILE, "utf-8")); } catch { return []; }

  await ensure();
  await fs.writeFile(FILE, JSON.stringify(items, null, 2));
}

async function addContent(item) {
  const all = await getAllContent();
  all.unshift(item); // newest first
  // Keep last 100 items
  if (all.length > 100) all.length = 100;
  await saveAll(all);
  return all;
}

async function deleteContent(id) {

  return filtered;
}

async function generateContent({ apiKey, prompt, platforms, tone, brandVoice, additionalContext, skills, onProgress }) {
  const client = new Anthropic({ apiKey });

  // Build skills context — summarize available skills for Claude
  const skillsSummary = skills.length > 0
    ? skills.map(s => `- **${s.name}** (${s.category}): ${s.summary}`).join("\n")
    : "No skills trained yet. Generate content based on general best practices.";

  // Detailed skill knowledge for Claude to draw from
  const skillDetails = skills.length > 0
    ? skills.map(s => `### ${s.name} (${s.category})
${s.description}
Key concepts: ${(s.key_concepts || []).join(", ")}
Use cases: ${(s.use_cases || []).join("; ")}
${s.code_example ? `Example:\n${s.code_example}` : ""}`).join("\n\n")
    : "";

  // Build platform requirements
  const platformDetails = platforms
    .map(p => PLATFORM_SPECS[p])
    .filter(Boolean)
    .map(p => `### ${p.name}\n${p.guidelines}`)
    .join("\n\n");

  const systemPrompt = `You are ATREYU, an elite AI content strategist and creator. You have been trained on specific skills from real codebases and knowledge bases. Use these skills to inform and enhance the content you create.

## YOUR TRAINED SKILLS:
${skillsSummary}

## DETAILED SKILL KNOWLEDGE:
${skillDetails}

## INSTRUCTIONS:
1. First, analyze which of your trained skills are most relevant to this content request
2. List the specific skills you are applying and WHY
3. Then generate the requested content for EACH platform specified
4. The content should be informed by your skill knowledge — not generic
5. Each platform's content must follow its specific format guidelines exactly

## OUTPUT FORMAT:
You MUST respond with ONLY a valid JSON object (no markdown fences, no explanation outside JSON) with this structure:
{
  "skills_used": [
    { "skill_name": "...", "reason": "Why this skill is relevant to this content" }
  ],
  "content": [
    {
      "platform": "platform-id",
      "platform_name": "Platform Display Name",
      "pieces": [
        {
          "type": "primary|variation|thread|slide|script",
          "label": "Main Post|Variation 1|Slide 1|Script|etc",
          "text": "The actual content text",
          "notes": "Additional notes, suggestions, or directions"
        }
      ],
      "hashtags": ["tag1", "tag2"],
      "metadata": {
        "suggested_visual": "Description of suggested visual/image",
        "best_posting_time": "Suggested posting time/day",
        "target_audience": "Who this content targets",
        "estimated_engagement": "Expected engagement level and why"
      }
    }
  ],
  "strategy_notes": "Overall content strategy explanation connecting the skills to the output"
}`;

  const userPrompt = `## CONTENT REQUEST:
${prompt}

## PLATFORMS TO CREATE FOR:
${platformDetails}

## TONE:
${tone}

${brandVoice ? `## BRAND VOICE:\n${brandVoice}` : ""}

${additionalContext ? `## ADDITIONAL CONTEXT:\n${additionalContext}` : ""}

Generate the content now. Remember to:
1. State which trained skills you're using and why
2. Create platform-specific content following each platform's guidelines exactly
3. Make the content informed by your skill knowledge, not generic
4. Return ONLY valid JSON`;

  onProgress({ stage: "generating", message: `Generating content for ${platforms.length} platform(s) using ${skills.length} trained skills...` });

  const response = await client.messages.create({
    model: "claude-sonnet-4-6",
    max_tokens: 12000,
    system: systemPrompt,
    messages: [{ role: "user", content: userPrompt }],
  });

  const text = response.content[0].text.trim();

  onProgress({ stage: "processing", message: "Processing generated content..." });

  let parsed;
  try {
    const jsonStr = text.replace(/^```json?\s*/, "").replace(/\s*```$/, "");
    parsed = JSON.parse(jsonStr);
  } catch (err) {
    throw new Error("Failed to parse content from Claude. Try again.");
  }

  // Wrap in a content item with metadata
  const contentItem = {
    id: uuidv4(),
    prompt,
    platforms,
    tone,
    brand_voice: brandVoice || "",
    skills_used: parsed.skills_used || [],
    content: parsed.content || [],
    strategy_notes: parsed.strategy_notes || "",
    created_at: new Date().toISOString(),
  };

  return contentItem;
}

// Platform selection
function togglePlatform(chip) {
  const platform = chip.dataset.platform;
  const idx = selectedPlatforms.indexOf(platform);
  if (idx === -1) {
    selectedPlatforms.push(platform);
    chip.classList.add("selected");
  } else {
    selectedPlatforms.splice(idx, 1);
    chip.classList.remove("selected");
  }
}

// Generate content
async function generateContent() {
  const prompt = document.getElementById("content-prompt").value.trim();
  const tone = document.getElementById("content-tone").value;
  const brandVoice = document.getElementById("brand-voice").value.trim();
  const additionalContext = document.getElementById("additional-context").value.trim();
  const btn = document.getElementById("generate-btn");
  const errEl = document.getElementById("generate-error");
  const prog = document.getElementById("gen-progress");
  const steps = document.getElementById("gen-steps");
  const fill = document.getElementById("gen-progress-fill");

  errEl.style.display = "none";

  if (!prompt) { showGenError("Describe what content you want to create."); return; }
  if (selectedPlatforms.length === 0) { showGenError("Select at least one platform."); return; }

  const authR = await fetch("/api/auth/status");
  const authD = await authR.json();
  if (!authD.authenticated) { showGenError("Connect your Anthropic API key first (go to IDE page)."); return; }

  btn.disabled = true;
  btn.querySelector(".btn-text").style.display = "none";
  btn.querySelector(".btn-loading").style.display = "inline";
  prog.style.display = "block";
  steps.innerHTML = "";
  fill.style.width = "5%";

  try {
    const r = await fetch("/api/content/generate", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ prompt, platforms: selectedPlatforms, tone, brandVoice, additionalContext }),
    });

    const reader = r.body.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      for (const line of decoder.decode(value, { stream: true }).split("\n")) {
        if (!line.startsWith("data: ")) continue;
        let event;
        try { event = JSON.parse(line.slice(6)); } catch { continue; }

        addGenStep(event.stage, event.message);

        switch (event.stage) {
          case "loading_skills": fill.style.width = "15%"; break;
          case "selecting_skills": fill.style.width = "30%"; break;
          case "generating": fill.style.width = "50%"; break;
          case "processing": fill.style.width = "80%"; break;
          case "saving": fill.style.width = "90%"; break;
          case "done":
            fill.style.width = "100%";
            currentOutput = event.content;
            renderOutput(event.content);
            loadHistory();
            setTimeout(() => { prog.style.display = "none"; fill.style.width = "0%"; }, 2000);
            break;
          case "error":
            fill.style.width = "0%";
            showGenError(event.message);
            break;
        }
      }
    }
  } catch (err) {
    showGenError("Generation failed: " + err.message);
    fill.style.width = "0%";
  } finally {
    btn.disabled = false;
    btn.querySelector(".btn-text").style.display = "inline";
    btn.querySelector(".btn-loading").style.display = "none";
  }
}

function addGenStep(stage, message) {
  const steps = document.getElementById("gen-steps");

    el.classList.remove("active");
    el.classList.add("done");

  const d = document.createElement("div");
  const isE = stage === "error", isD = stage === "done";

  d.innerHTML = `<span class="step-icon">${isE ? "\u2717" : isD ? "\u2713" : "\u25CF"}</span><span>${esc(message)}</span>`;
  steps.appendChild(d);
}

              ${platform.metadata.target_audience ? `<div class="meta-item"><div class="meta-label">Target Audience</div><div class="meta-value">${esc(platform.metadata.target_audience)}</div></div>` : ""}
              ${platform.metadata.estimated_engagement ? `<div class="meta-item"><div class="meta-label">Engagement Estimate</div><div class="meta-value">${esc(platform.metadata.estimated_engagement)}</div></div>` : ""}
            </div>
          ` : ""}
        </div>
      </div>
    `;
  }).join("");

  // Scroll to output
  section.scrollIntoView({ behavior: "smooth", block: "start" });
}

// Copy content
function copyPiece(btn, id) {
  const el = document.getElementById(id);
  // Get text without the copy button text
  const text = el.innerText.replace("Copy", "").replace("Copied!", "").trim();
  navigator.clipboard.writeText(text).then(() => {
    btn.textContent = "Copied!";
    btn.classList.add("copied");
    setTimeout(() => {
      btn.textContent = "Copy";
      btn.classList.remove("copied");
    }, 2000);
  });
}

// History
async function loadHistory() {
  try {
    const r = await fetch("/api/content/history");
    const history = await r.json();
    const list = document.getElementById("history-list");
    const count = document.getElementById("history-count");
    count.textContent = history.length ? `${history.length} items` : "";

    if (history.length === 0) {
      list.innerHTML = '<div style="color:var(--text-muted);font-size:13px;padding:20px 0;text-align:center">No content generated yet</div>';
      return;
    }

    const platformIcons = {
      "instagram-post": "&#128247;", "instagram-story": "&#128248;", "instagram-reel": "&#127910;",
      "youtube-short": "&#9654;", "youtube-video": "&#127909;", "twitter-x": "&#128172;",
      "linkedin": "&#128188;", "tiktok": "&#127925;", "blog-post": "&#128221;",
      "email-copy": "&#9993;", "ad-copy": "&#9889;", "website-copy": "&#127760;"
    };

    list.innerHTML = history.map(item => `
      <div class="history-item" onclick="loadHistoryItem('${item.id}')">
        <span class="history-prompt">${esc(item.prompt)}</span>
        <span class="history-platforms">${(item.platforms || []).map(p => platformIcons[p] || "").join(" ")}</span>
        <span class="history-date">${new Date(item.created_at).toLocaleDateString()}</span>
        <button class="history-delete" onclick="event.stopPropagation();deleteHistoryItem('${item.id}')">Delete</button>
      </div>
    `).join("");
  } catch {}
}

async function loadHistoryItem(id) {
  try {
    const r = await fetch(`/api/content/${id}`);
    const item = await r.json();
    renderOutput(item);
  } catch {}
}

async function deleteHistoryItem(id) {
  if (!confirm("Delete this content?")) return;
  await fetch(`/api/content/${id}`, { method: "DELETE" });
  loadHistory();
}

function esc(s) {
  if (!s) return "";
  const d = document.createElement("div");
  d.textContent = s;
  return d.innerHTML;
}
